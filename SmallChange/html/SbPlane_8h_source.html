51  SbPlane(const SbVec3<Type> & p0, const SbVec3<Type> & p1, const SbVec3<Type> & p2) {
52  this->normal = (p1 - p0).cross(p2 - p0);
53  assert(this->normal.sqrLength() != 0.0f); // points on line
54  this->normal.normalize();
